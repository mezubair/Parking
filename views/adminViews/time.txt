vehicleSchema.pre('save', function (next) {
  if (this.inTime && this.outTime) {
    const timeDifference = this.outTime - this.inTime;
    const hoursParked = timeDifference / (1000 * 60 * 60); // Convert milliseconds to hours
    const ratePerHour = 10; // Change this to your rate per hour

    this.totalCharge = hoursParked * ratePerHour;
  }
  next();
});
